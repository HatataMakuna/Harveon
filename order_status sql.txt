CREATE TABLE ORDER_STATUS (
 ORDER_DETAILS_ID INT PRIMARY KEY GENERATED ALWAYS AS IDENTITY,
 ORDER_ID INT REFERENCES ORDER(ORDER_ID),
 CUSTOMER_ID REFERENCES CUSTOMER(CUSTOMER_ID),
 DT DATE AS DATE,
 FINAL_PRICE DOUBLE,
 STATUS VARCHAR(20) DEFAULT 'Packaging'
);
